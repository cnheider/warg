from typing import Any, Mapping, Tuple



[docs]def map_value_product(dicts: Mapping) -> Any:
    """ """
    return (dict(zip(dicts, x)) for x in itertools.product(*dicts.values()))




[docs]def map_product(dicts: Mapping, repeat: int = 2) -> Any:
    """ """
    yield from zip(
        itertools.product(dicts.keys(), repeat=repeat),
        itertools.product(dicts.values(), repeat=repeat),
    )




[docs]def map_permutations(dicts: Mapping, repeat: int = 2) -> Tuple:
    """ """
    yield from zip(
        itertools.permutations(dicts.keys(), repeat),
        itertools.permutations(dicts.values(), repeat),
    )

    """ """
