[docs]def color_from_str(s: str, seperator: str = " ") -> COLOR_INT:
    """

    :param s:
    :type s:
    :param seperator:
    :type seperator:
    :return:
    :rtype:"""
    components = s.split(seperator)
    n = len(components)
    if n == 3:
        return RGB(*[int(c) for c in components])
    elif n == 4:
        return RGBA(*[int(c) for c in components])
    raise NotImplementedError("Color space not recognised")




[docs]def color_to_str(t: COLOR_INT, seperator: str = " ") -> str:
    """

    :param t:
    :type t:
    :param seperator:
    :type seperator:
    :return:
    :rtype:"""
    return seperator.join([str(c) for c in t])
